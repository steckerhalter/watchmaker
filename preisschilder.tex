\newcommand{\uhrschild}[3]{%

        % Schneidelinie / Rahmen
        \draw[thick, dashed, gray] (0,0) rectangle (8.5,5.5);

        \draw[thin, black] (0.2,0.2) rectangle (8.3,5.3);

        \node[text width=7.5cm, align=center] at (4.25,4.3) {
            \fontsize{16}{18}\selectfont\textbf{#1}
        };
        
        % Trennlinie
        \draw[thin, gray] (1,3.2) -- (7.5,3.2);
        
        % Zusatzinfo (z.B. Herkunft, Werk, Zustand)
        \node[text width=7.5cm, align=center, text=black!70] at (4.25,2.5) {
            \fontfamily{cmss}\selectfont\large #2
        };

        \node[draw, thick, rounded corners=3pt, fill=black!5, inner sep=6pt] at (4.25,1.0) {
            \fontsize{20}{22}\selectfont\textbf{CHF #3}
        };
    \end{tikzpicture}
    \vspace{0.4cm}
}
