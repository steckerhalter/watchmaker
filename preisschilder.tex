% Makro für die Erstellung eines einzelnen Uhren-Schildes
\newcommand{\uhrschild}[5]{%
    \begin{tikzpicture}[baseline=(current bounding box.center)]
        % Gestrichelte Außenlinie als Schneidehilfe
        \draw[dashed, line width=0.5pt, gray] (0,0) rectangle (8.5,5.2);
        % Innerer, dicker Rahmen für die edle Optik
        \draw[line width=1.2pt, black] (0.2,0.2) rectangle (8.3,5.0);
        
        % Kopfzeile mit Marke / Modell
        \node[anchor=north west, font=\bfseries\large, text width=7.5cm] at (0.4,4.8) {#1};
        \draw[line width=0.8pt, black] (0.4,4.2) -- (8.1,4.2);
        
        % Uhren-Details
        \node[anchor=north west, font=\small, text width=7.5cm] at (0.4,4.0) {
            \textbf{Typ:} #2 \\
            \textbf{Baujahr:} #3 \\
            \textbf{Besonderheiten:} #4
        };
        
        % Gepunktete Trennlinie über dem Preis
        \draw[dotted, line width=0.5pt, gray] (0.4,1.5) -- (8.1,1.5);
        
        % Preis-Anzeige unten rechts
        \node[anchor=south east, font=\bfseries\LARGE] at (8.1,0.4) {CHF #5.-};
        % Kleiner Qualitätsstempel unten links
        \node[anchor=south west, font=\itshape\tiny, gray] at (0.4,0.4) {Geprüfte Qualität};
    \end{tikzpicture}%
}
